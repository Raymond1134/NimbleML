# toyGPT — run from repo root

## Prepare data (one-time: download FineWeb-Edu, train BPE, encode -> token bins)
python toyGPT\prepare_data.py
  (cached by content hash; safe to re-run, train/resume reuse it instantly)

## Train
python toyGPT\train_gpt.py
  (auto-prepares data if the cache is missing)

## Resume
python toyGPT\train_gpt.py --resume latest
python toyGPT\train_gpt.py --resume best
python toyGPT\train_gpt.py --resume step_1000

## Playground (Nova chatbot, loads latest by default)
python toyGPT\playground.py
python toyGPT\playground.py --checkpoint latest --temperature 0.8 --top-k 0 --len 200

  :clear   :temp 0.8   :topk 40   :len 300   :ckpt latest   :quit
  (anything else = your message to Nova)

## One-shot generate
python toyGPT\generate.py --checkpoint best --prompt "The"
python toyGPT\generate.py --checkpoint best --temperature 0.8 --top-k 40 --chars 300

## Paths
  config:      toyGPT\gpt_toy_config.toml
  checkpoints: toyGPT\checkpoints\best | latest | step_N
  tokenizer:   toyGPT\data\tokenizer.json
  token bins:  toyGPT\data\cache\encoded\fineweb-edu\<hash>\{train,val}.bin (uint16 memmap)

## Troubleshooting CUDA / NaN
  - Use batch_size=8 in gpt_toy_config.toml (not 10+ on 8 GB VRAM).
  - After any CUDA crash, start a fresh terminal/Python process (GPU context is poisoned).
  - Starting loss should be ~9–10, not ~300 (embedding init is 0.02 std).
  - Training skips steps with non-finite loss/grad and logs a warning.

## Data settings (gpt_toy_config.toml [data])
  hf_subset = "sample-10BT"     # FineWeb-Edu shard set to stream from
  train_tokens / val_tokens     # corpus size budget (2.6B train by default)
  tokenizer_sample_mb           # text streamed to train the BPE vocab
